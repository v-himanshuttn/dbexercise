create table mycustomer
(
 custid int PRIMARY KEY NOT NULL,
 cname  varchar(100)
)

----------------------------------------------------------------------------------
create table mysalesman
(
 salesid int PRIMARY KEY NOT NULL,
 sname  varchar(100)
)
-----------------------------------------------------------------------------------

create table myorders
(
 orderid int,
 PRIMARY KEY(orderid),
 cid int,
 sid int,
 FOREIGN KEY (cid) REFERENCES mycustomer (custid),  
 FOREIGN KEY (sid) REFERENCES mysalesman (salesid)  
 )
